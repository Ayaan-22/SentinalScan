Sub-step 2 — Async crawl_page with AsyncClient (4 hrs)

Convert crawl_page() to async def and switch to httpx.AsyncClient. Keep VulnerabilityTester fully synchronous. Run the async crawler from a sync context via asyncio.run() as a temporary bridge.
backend/crawler_async.py
import httpx, asyncio
from selectolax.parser import HTMLParser

class AsyncWebCrawler:
    async def __aenter__(self):
        self.client = httpx.AsyncClient(
            headers={"User-Agent": self.config.user_agent},
            verify=self.config.verify_ssl,
            timeout=self.config.timeout,
            limits=httpx.Limits(
                max_connections=self.config.workers * 2,
                max_keepalive_connections=self.config.workers,
            ),
            follow_redirects=True,
        )
        self.response_cache: dict[str, str] = {}
        return self

    async def __aexit__(self, *args):
        await self.client.aclose()

    def _extract_links(self, url: str, html: str) -> set[str]:
        """selectolax is 5-10x faster than BeautifulSoup for this."""
        tree = HTMLParser(html)
        links = set()
        for node in tree.css("a[href], form[action]"):
            href = (node.attributes.get("href") or
                    node.attributes.get("action", ""))
            if href and self._is_valid_link(url, href):
                full = urljoin(url, href).split("#")[0]
                links.add(full)
        return links

    async def crawl_page(self, url: str):
        self.rate_limiter.wait()
        try:
            r = await self.client.get(url)
            r.raise_for_status()
            self.response_cache[url] = r.text
            links = self._extract_links(url, r.text)
            self.logger.info(f"Crawled: {url} ({len(links)} links)")
            return url, links, True
        except Exception as e:
            self.logger.debug(f"Crawl failed {url}: {e}")
            return url, set(), False

    async def crawl(self) -> list[str]:
        queue = asyncio.Queue()
        await queue.put(self.base_url)
        visited, pages = set(), []
        sem = asyncio.Semaphore(self.config.workers)

        async def worker():
            while True:
                url = await queue.get()
                try:
                    if url in visited or len(visited) >= self.config.max_pages:
                        return
                    visited.add(url)
                    async with sem:
                        _, links, ok = await self.crawl_page(url)
                    if ok:
                        pages.append(url)
                        for lnk in links:
                            if lnk not in visited:
                                await queue.put(lnk)
                finally:
                    queue.task_done()

        workers = [asyncio.create_task(worker())
                   for _ in range(self.config.workers)]
        await queue.join()
        for w in workers: w.cancel()
        await asyncio.gather(*workers, return_exceptions=True)
        return pages


# Temporary sync bridge (removed in sub-step 4)
def crawl_sync(config, logger) -> list[str]:
    async def _run():
        async with AsyncWebCrawler(config, logger) as c:
            return await c.crawl()
    return asyncio.run(_run())

Sub-step 3 — Async vulnerability tester (6 hrs)

Convert all test methods to async def. Replace time.sleep() with await asyncio.sleep() in time-based SQLi payloads. Use asyncio.gather() to run all tests for a page concurrently — this collapses per-page test time from worst-case-sum to worst-case-single.
backend/tester_async.py
class AsyncVulnerabilityTester:

    async def test_page(self, url: str,
                          cached_html: str | None = None):
        # Reuse crawl cache — no redundant HTTP request
        if cached_html is None:
            r = await self.client.get(url)
            cached_html, response = r.text, r
        else:
            response = MockResponse(cached_html)

        forms = self._extract_forms(url, cached_html)
        tasks = [
            self.test_open_redirect(url),
            self.test_path_traversal(url),
            self.test_info_disclosure(url, cached_html),
            self.test_security_headers(url, response),
            self.test_cookie_security(url, response),
            self.test_sensitive_files(url),
            *[self._test_form(f, url) for f in forms],
        ]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        vulns = []
        for r in results:
            if isinstance(r, list): vulns.extend(r)
            elif isinstance(r, Vulnerability): vulns.append(r)
            elif isinstance(r, Exception):
                self.logger.warning(f"Test error on {url}: {r}")
        return vulns

    async def _test_form(self, form, url: str) -> list[Vulnerability]:
        results = await asyncio.gather(
            self.test_xss(form, url),
            self.test_sqli(form, url),
            self.test_csrf(form, url),
            return_exceptions=True
        )
        return [r for r in results if isinstance(r, Vulnerability)]

    async def test_sqli(self, form, url: str):
        # Baseline timing — async, does not block event loop
        t0 = asyncio.get_event_loop().time()
        await self._submit_form(form, url, "baseline_val")
        baseline_time = asyncio.get_event_loop().time() - t0

        # Error-based (fast — no sleep)
        for payload, desc in self._SQLI_ERROR_PAYLOADS:
            r = await self._submit_form(form, url, payload)
            if r and self._has_sql_error(r.text):
                return self._make_vuln(VulnType.SQLI, url, desc)

        # Time-based — await asyncio.sleep() instead of time.sleep()
        # This releases the event loop while waiting, allowing other
        # tests to run concurrently
        for payload in self._SQLI_TIME_PAYLOADS:
            t1 = asyncio.get_event_loop().time()
            await self._submit_form(form, url, payload)
            elapsed = asyncio.get_event_loop().time() - t1
            if elapsed > baseline_time + MIN_TIME_DELAY_DETECTION:
                return self._make_time_based_vuln(url, elapsed, baseline_time)
        return None

    async def test_sensitive_files(self, url: str) -> list[Vulnerability]:
        # 8 files probed concurrently instead of sequentially
        tasks = [self._probe_file(url, f) for f in self._SENSITIVE_FILES]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        return [r for r in results if isinstance(r, Vulnerability)]

Sub-step 4 — Full async pipeline, remove ThreadPoolExecutor (4 hrs)

Wire VulnerabilityScanner.scan() to asyncio.run(). Remove all manual thread management. The Celery task wrapper becomes the only threading boundary in the system.
backend/vuln_scanner.py — final async scan()
class VulnerabilityScanner:

    async def scan(self) -> list[Vulnerability]:
        self._verify_authorization()
        self._validate_target()
        self.start_time = asyncio.get_event_loop().time()
        self.logger.info("="*60)
        self.logger.info("VULNERABILITY SCAN STARTED")
        self.logger.info(f"Target: {self.config.target_url}")
        self.logger.info("="*60)
        await self._phase_crawl()
        await self._phase_test()
        self._deduplicate_vulnerabilities()
        self._emit_summary()
        return self.vulnerabilities

    async def _phase_crawl(self):
        self.logger.info("[PHASE 1] Crawling")
        async with AsyncWebCrawler(self.config, self.logger) as crawler:
            self.pages = await crawler.crawl()
            self._page_cache = crawler.response_cache
        self.logger.info(f"Crawl complete: {len(self.pages)} pages")

    async def _phase_test(self):
        self.logger.info(f"[PHASE 2] Testing {len(self.pages)} pages")
        async with httpx.AsyncClient(
            verify=self.config.verify_ssl,
            timeout=self.config.timeout,
        ) as client:
            tester = AsyncVulnerabilityTester(client, self.config, self.logger)
            coros = [
                tester.test_page(p, self._page_cache.get(p))
                for p in self.pages
                if self._scan_active
            ]
            results = await asyncio.gather(*coros, return_exceptions=True)
            for r in results:
                if isinstance(r, list):
                    self.vulnerabilities.extend(r)

    def _emit_summary(self):
        duration = asyncio.get_event_loop().time() - self.start_time
        self.logger.info(f"Scan complete in {duration:.1f}s — "
                         f"{len(self.vulnerabilities)} findings")

Prioritization strategy
With zero tests today, write in risk order, not coverage order. An SSRF validation test that catches a critical security regression is worth 50 UI snapshot tests. Target: 70% line coverage before Phase 3 begins.
T1

Tier 1 — Highest regression risk (write first)

1
SSRF blocklist validation
Most critical. Verifies the security boundary cannot be bypassed. If these tests regress, the tool becomes a weapon against internal infrastructure.
2
Scan state lifecycle — is_scanning always resets
The original bug: exceptions leave is_scanning=True forever. This test verifies the try/finally guard works under failure conditions.
3
Deduplication correctness
Without dedup tests, a refactor could silently produce 10× the expected findings. Test both identical-fingerprint removal and URL-distinct preservation.
4
URL parsing edge cases
Malformed URLs, IPv6 addresses, and URLs with embedded credentials can all bypass the blocklist if not explicitly tested.

def test_scan_active_false_after_exception():
    """is_scanning must NEVER remain True on scan failure."""
    session = ScanSession(scan_id="test-id", is_scanning=True)
    try:
        raise RuntimeError("simulated crash")

        pass

def test_dedup_removes_identical_fingerprints():

Tier 2 — Mock-based HTTP detection tests

Use respx to intercept httpx calls at the transport layer. No network required — safe in CI. Test both the positive case (vulnerability detected) and the negative case (no false positive on clean input).

Security headers middleware

Removes stack fingerprinting headers (Server, X-Powered-By) and adds defensive response headers on every API response. Register after the CORS middleware in api.py.
backend/middleware.py
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
import re, html

class SecurityHeadersMiddleware(BaseHTTPMiddleware):
    """Add defensive headers and strip stack fingerprints."""
    async def dispatch(self, request: Request, call_next):
        response = await call_next(request)
        h = response.headers
        h["X-Content-Type-Options"] = "nosniff"
        h["X-Frame-Options"]       = "DENY"
        h["Referrer-Policy"]       = "no-referrer"
        h["Permissions-Policy"]    = "camera=(), microphone=(), geolocation=()"
        h["X-XSS-Protection"]      = "0"  # modern browsers: disable the broken XSS auditor
        h.pop("server", None)         # removes "uvicorn" fingerprint
        h.pop("x-powered-by", None)
        return response


class SensitiveDataSanitizer(BaseHTTPMiddleware):
    """Mask secrets from any data that might appear in logs."""
    _MASK = re.compile(
        r'("(?:auth_token|password|cookie|secret)[^"]*":\s*")([^"]+)(")',
        re.IGNORECASE)

    @classmethod
    def sanitize(cls, text: str) -> str:
        return cls._MASK.sub(r'\1[REDACTED]\3', text)

    async def dispatch(self, request: Request, call_next):
        return await call_next(request)

Log payload sanitization

XSS payloads flow from the scanner into the WebSocket log stream. Sanitize HTML special characters before pub/sub to prevent reflected payloads appearing in the operator's browser. This is defense-in-depth — React's JSX currently escapes these, but sanitizing at the source removes the risk permanently.
backend/logger.py
import html, re, asyncio, time, os

_DANGEROUS = re.compile(r'[<>&"\'`]')

def sanitize_log_message(msg: str) -> str:
    """Escape HTML-special chars from scan payloads before streaming."""
    return _DANGEROUS.sub(
        lambda m: html.escape(m.group(), quote=True), msg)


class RedisPublishLogger:
    """Async-safe logger that publishes to Redis pub/sub."""

    def __init__(self, scan_id: str):
        self.scan_id = scan_id
        self.verbose = True

    def _log(self, level: str, msg: str):
        safe_msg = sanitize_log_message(str(msg))
        entry = {
            "timestamp": time.time(),
            "level": level,
            "message": safe_msg,
            "scan_id": self.scan_id,
        }
        # Publish without blocking the calling thread
        if loop is not None and not loop.is_closed():
            asyncio.run_coroutine_threadsafe(
                store.publish_log(self.scan_id, entry), loop
            )

    def info(self, msg):    self._log("INFO", msg)
    def warning(self, msg): self._log("WARNING", msg)
    def error(self, msg):   self._log("ERROR", msg)
    def debug(self, msg):   self._log("DEBUG", msg)
    def critical(self, msg): self._log("CRITICAL", msg)

## Architecture decisions

**Why httpx over requests** — identical API surface but adds native async
and HTTP/2. Required prerequisite for the concurrent scanning pipeline.

**Why Celery over threading** — decouples HTTP from compute. The API
returns a `scan_id` in under 50ms while a worker picks up the actual
scan. Horizontal scaling requires zero API changes.

**Why Redis pub/sub for logs** — scan logs broadcast to WebSocket
subscribers without per-connection state. Multiple browser tabs receive
the same log stream without any sync logic in the API layer.

**Why UUID session-keyed state** — the original design used a global
singleton that allowed only one concurrent scan and left `is_scanning=true`
permanently on exceptions. UUID keys enable concurrent scans, proper
cleanup on failure, and result persistence across API restarts.

## Security hardening decisions

Three critical vulnerabilities were identified in the original design
and fixed as Phase 1 of the upgrade:

**SSRF via target URL** — the scanner accepted any URL including internal
network addresses, allowing it to probe `http://192.168.1.x` or
`http://169.254.169.254` (AWS metadata). Fixed by hard-rejecting all
RFC-1918 and cloud metadata CIDR ranges before any HTTP request fires.

**No API authentication** — `POST /scan/start` accepted requests from
any origin. Fixed by requiring an `X-API-Key` header validated against
a server environment variable via FastAPI `Depends()`.

**Irrecoverable scan state** — exceptions in the scan thread left
`is_scanning=true` permanently; the only recovery was restarting the
process. Fixed by wrapping all scan logic in `try/finally` that always
resets the session state.
